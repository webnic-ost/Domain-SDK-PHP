"administratorContactId" => "WN964984T", // The contact ID for the administrator.

"billingContactId" => "WN964984T", // The contact ID for billing information.

"ns2.web.cc"
],
"subscribeProxy" => false, // Indicates if the domain should be subscribed to a proxy.

]
